Pregunta 2

Mediante un archivo XML, analizar dicho archivo XML utilizando el siguiente Script:

xmlhttp = new XMLHttpRequest();
xmlhttp.open("GET", "ejemplo.xml", false);
xmlhttp.send();
xmlDoc = xmlhttp.responseXML;

Solucion

Codigo 


<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<meta name="viewport" content="width=device-width, initial-scale=1">
<title>Pregunta2</title>                    
</head>
<body>
<p class="text-center" id="demo"></p>
<button type="button" class="btn btn-danger" onclick="myFunction()">Traer XML</button>            
<script>
  function myFunction() {
    var xhttp = new XMLHttpRequest();
    xhttp.onreadystatechange = function() {
      if (this.readyState == 4 && this.status == 200) {
        obtenerDatos(this);
        }else{
        document.getElementById("demo").innerHTML = "Error al cargar los datos";
        }
      };
    xhttp.open("GET", "archivo2.xml", false);
    xhttp.send();
  } 
                    
  function obtenerDatos(xml) {
    var xmlDoc = xml.responseXML;
      document.getElementById("demo").innerHTML = 
      xmlDoc.getElementsByTagName("title")[0].childNodes[0].nodeValue + "<br>" +
      xmlDoc.getElementsByTagName("author")[0].childNodes[0].nodeValue + "<br>" +
      xmlDoc.getElementsByTagName("year")[0].childNodes[0].nodeValue;
  }
</script>
</body>
</html>
