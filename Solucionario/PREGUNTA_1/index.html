Pregunta 1

Analizar una cadena de texto en un objeto DOM XML y extraer esa Información con JavaScript

Solucion

Codigo 


<!DOCTYPE html>
<html>
<body>
                    
<p id="pregunta1"></p>
<button id="boton">Pulse aqui!</button>
<p>Extraer el titulo de un XML </p>
<script>
  var parser, xmlDoc;
  var text = 
    "<bookstore><book>" +
    "<title>Grupo06</title>" +
    "<author>Cabezas - Chanchay</author>" +
    "<year>2022</year>" +
    "</book></bookstore>";
                    
   document.querySelector('#boton').addEventListener('click',function(){                
    parser = new DOMParser();
    xmlDoc = parser.parseFromString(text,"text/xml");
    document.getElementById("pregunta1").innerHTML =
    xmlDoc.getElementsByTagName("title")[0].childNodes[0].nodeValue;
  });
                    
<<script>
                    
</body>
</html>
