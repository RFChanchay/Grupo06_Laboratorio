Pregunta 3

Desarrollar una tabla HTML según el valor de un menú desplegable, utilizando la función de JavaScript JSON.parse() para convertir texto en un objeto de JavaScript.

Solucion

Codigo 


                    
<!DOCTYPE html>
<html>
<head>
<title>P3</title>
<style>
select, p, input {
font: 1em Calibri;
}
</style>
</head>
<body>
<p>
<input type="button" 
style="margin:10px 0;" 
onclick="populateSelect()" 
value="Click para cargar el texto en el menu" />
</p>

                    
<!--The SELECT element.-->
<select id="sel" onchange="show(this)">
<option value="">-- Selecciona un ave --</option>
</select>

                    
<p id="msg"></p>

                    

</body>

                    
<script>
function populateSelect() {

var pajaritos='[{"ID": "001","Bird_Name": "condor"},{"ID": "002","Bird_Name": "tucan"},{"ID": "003","Bird_Name": "aguila"}]';
var pajaros = JSON.parse(pajaritos);

                    
var ele = document.getElementById('sel');
for (var i = 0; i < pajaros.length; i++) {
ele.innerHTML = ele.innerHTML +
'<option value="' + pajaros[i]['ID'] + '">' + pajaros[i]['Bird_Name'] + '</option>';
}
}
function show(ele) {
// Mostrar inf.
var msg = document.getElementById('msg');
msg.innerHTML = 'Pajaro seleccionado: <b>' + ele.options[ele.selectedIndex].text + '</b> </br>' +
'ID: <b>' + ele.value + '</b>';
}

                    

</script>
</html>
